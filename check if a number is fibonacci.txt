#include <stdio.h>

int main() {
    int num, a = 0, b = 1, next = 0;

    printf("Enter a number: ");
    scanf("%d", &num);

    while (next < num) {
        next = a + b;
        a = b;
        b = next;
    }

    if (next == num || num == 0 || num == 1)
        printf("%d is a Fibonacci number\n", num);
    else
        printf("%d is NOT a Fibonacci number\n", num);

    return 0;
}
