#include <stdio.h>

int main() {
    int n, a = 0, b = 1, next, count = 1;

    printf("Enter number of terms: ");
    scanf("%d", &n);

    printf("Fibonacci Series: ");

    while (count <= n) {
        printf("%d ", a);
        next = a + b;
        a = b;
        b = next;
        count++;
    }

    return 0;
}
