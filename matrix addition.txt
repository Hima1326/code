#include <stdio.h>

int main() {
    int a[10][10], b[10][10], res[10][10];
    int r, c, i, j;

    printf("Enter rows and columns: ");
    scanf("%d %d", &r, &c);

    printf("Enter Matrix A:\n");
    for (i=0; i<r; i++)
        for (j=0; j<c; j++)
            scanf("%d", &a[i][j]);

    printf("Enter Matrix B:\n");
    for (i=0; i<r; i++)
        for (j=0; j<c; j++)
            scanf("%d", &b[i][j]);

    for (i=0; i<r; i++)
        for (j=0; j<c; j++)
            res[i][j] = a[i][j] + b[i][j];

    printf("Result (A + B):\n");
    for (i=0; i<r; i++) {
        for (j=0; j<c; j++)
            printf("%d ", res[i][j]);
        printf("\n");
    }

    return 0;
}
